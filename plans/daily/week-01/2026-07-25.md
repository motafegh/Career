python - <<'PY'
import json
from pathlib import Path

required = {
    "schema_version",
    "observed_at",
    "environment",
    "source",
    "service",
    "event_type",
    "outcome",
    "auth_method",
    "principal",
    "remote_address",
    "remote_port",
    "process_id",
    "raw_record",
    "parse_warnings",
}

for path, expected_outcome, expected_count in (
    (Path("/tmp/aegislab-week01/week01-success-final.jsonl"), "success", None),
    (Path("/tmp/aegislab-week01/week01-failure-final.jsonl"), "failure", 5),
):
    rows = [json.loads(line) for line in path.read_text(encoding="utf-8").splitlines()]
    if expected_count is not None and len(rows) != expected_count:
        raise SystemExit(f"{path}: expected {expected_count} rows, got {len(rows)}")
    if not rows:
        raise SystemExit(f"{path}: no rows")
    for number, row in enumerate(rows, 1):
        if set(row) != required:
            raise SystemExit(f"{path}:{number}: wrong keys: {set(row) ^ required}")
        if row["outcome"] != expected_outcome:
            raise SystemExit(f"{path}:{number}: wrong outcome")
    print(path.name, "PASS", len(rows))
PY
